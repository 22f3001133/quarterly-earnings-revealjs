Return on Equity (ROE):

\[ ROE = \frac{\text{Net Income}}{\text{Shareholders' Equity}} \]

Net Interest Margin (NIM):

\[ NIM = \frac{\text{Interest Income} - \text{Interest Expense}}{\text{Average Earning Assets}} \]

import pandas as pd

def compute_growth(df):
    df = df.sort_values('quarter')
    df['qoq_growth'] = df['revenue'].pct_change() * 100
    return df

data = pd.DataFrame({
    "quarter": ["Q1", "Q2", "Q3", "Q4"],
    "revenue": [95, 100, 110, 120]
})

print(compute_growth(data))
