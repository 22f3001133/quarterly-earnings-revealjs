import pandas as pd

def compute_qoq_growth(df):
    """
    df columns: ['quarter', 'revenue']
    Returns df with an added 'qoq_growth' percentage column.
    """
    df = df.sort_values('quarter')
    df['qoq_growth'] = df['revenue'].pct_change() * 100
    return df

data = pd.DataFrame({
    "quarter": ["Q1", "Q2", "Q3", "Q4"],
    "revenue": [95.0, 100.0, 110.0, 120.0]
})

result = compute_qoq_growth(data)
print(result)
